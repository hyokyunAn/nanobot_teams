nanobot run commands (env-driven)
=================================

1) one-time install
-------------------
cd /Users/ahk/github_codes/nanobot
python -m pip install -e .
python -m pip install -e ".[ui]"
python -m pip install -r /Users/ahk/github_codes/nanobot/16.proactive-messages/requirements.txt

cd /Users/ahk/github_codes/nanobot/16.proactive-messages
cp .env.example .env
# edit .env and fill required keys


2) relay run (backend server)
-----------------------------
cd /Users/ahk/github_codes/nanobot
export INTERNAL_TOKEN="<shared_token>"
export TEAMS_INTERNAL_TOKEN="<shared_token>"
export RELAY_BIND_HOST="<relay_bind_host>"
export RELAY_BIND_PORT="<relay_bind_port>"
export RELAY_INBOUND_TIMEOUT="<relay_inbound_timeout_sec>"
export TEAMS_PROACTIVE_URL="<teams_proactive_endpoint>"

python -m nanobot relay \
  --host "${RELAY_BIND_HOST}" \
  --port "${RELAY_BIND_PORT}" \
  --inbound-timeout "${RELAY_INBOUND_TIMEOUT}" \
  --teams-proactive-url "${TEAMS_PROACTIVE_URL}"
