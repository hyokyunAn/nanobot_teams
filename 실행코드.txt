3) teams production structure
-----------------------------
# backend server: run relay
cd /Users/ahk/github_codes/nanobot
export INTERNAL_TOKEN="shared_token"
export TEAMS_INTERNAL_TOKEN="shared_token"
python -m nanobot relay \
  --host 0.0.0.0 \
  --port 18800 \
  --inbound-timeout 120 \
  --teams-proactive-url "https://<your-appservice>/internal/proactive"
